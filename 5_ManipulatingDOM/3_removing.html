//Utilizando remove() para remover itens
//o codigo abaixo funciona apenas para itens que já estavam presentes no carregamento da pagina 

$('#places-container .remove').on('click', function(){
    $(this).parent().remove();
});
//Para o funcionamento correto, ao invés deste código devemos utilizar o código abaixo (utilizando delegation)
//para que os itens adicionados posteriormente possam ser removidos também 
$('#places-container').on('click', '.remove', function(){
    $(this).parent().remove();
});
