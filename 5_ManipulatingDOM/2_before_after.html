//Utilizando Before (para anexar antes do elmento) e After (para anexar após elmento)
//o texto digitado no input à div places-container 

$('#add-container').on('click','button',function(){
    var texto = $('#add-container input').val();
    var html = `<div class="item">
        <div class="remove">X</div>
        `+texto+`
    </div>`;
    if($(this).attr('class') == 'before'){
        $('#places-container').children().first().before(html)
    }else{
        $('#places-container').children().last().after(html);
    }
});
