//Utilizando Append (para anexar ao fim) e Prepend (para anexar no inicio)
//o texto digitado no input Ã  div places-container 

$('#add-container').on('click','button',function(){
    var texto = $('#add-container input').val();
    var html = `<div class="item">
        <div class="remove">X</div>
        `+texto+`
    </div>`;
    if($(this).attr('class') == 'append') {
        //Equivale a $(html).appendTo('#places-container');
        $('#places-container').append(html);

    }else{
        //Equivale a $(html).prependTo('#places-container');
        $('#places-container').prepend(html);
    }
});
