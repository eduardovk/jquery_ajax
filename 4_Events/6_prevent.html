//Cancelando o comportamento padr√£o (redirecionar) ao clicar em um link
//utilizando parametro e na function() do evento click e usando e.preventDefault() //e.stopPropagation() para evitar que o alerta seja exibido quando clicar apenas na div 

$('#caixa').on('click','a',function(e){
    e.preventDefault();
    e.stopPropagation();
    alert('Evento de redirecionamento de pagina cancelado');
});

$('#caixa').on('click', function(e){
    alert('Clicou na caixa!');
});
