//Exemplos de toggle(), fadeToggle() e slideToggle()
//com parametro de duração da animação e função anonima na conclusão 

$('.box').on('click', 'button', function(){
    var classe = $(this).attr('class');
    var texto = $(this).closest('#caixa').find('.more');
    if(classe === "toggle"){
        texto.toggle(500, function(){alert('Animação concluída.')});
    }else if(classe === "fade"){
        texto.fadeToggle(500, function(){alert('Animação concluída.')});
    }else{
        texto.slideToggle(500, function(){alert('Animação concluída.')});
    }
});
