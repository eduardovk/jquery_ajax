//Exibindo mensagem de loading com beforeSend
//e mensagem de requisição completada com metodo always() 

$('#caixa').on('click', function(){
    $.ajax('1_result.html', {
        beforeSend: function(){
            $('#status').text('Loading...');
        }
    }).done(function(response){
        $('#resultado').html(response);
    }).always(function(){
        $('#status').text('Completed');
    });
});
