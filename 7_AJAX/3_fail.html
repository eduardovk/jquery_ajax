//Simulando captura de arquivo não encontrado
//com método fail() 

$('#caixa').on('click', function(){
    $.ajax('inexistente.html', {
        beforeSend: function(){
            $('#status').text('Loading...');
        }
    }).fail(function(request, errorType, errorMessage){
        alert('Erro tipo: ' + errorType +'\nMensagem: ' + errorMessage);
    }).always(function(){
        $('#status').text('Completed');
    });
});
